from scipy import stats
import matplotlib.pyplot as plt
x=[0,6,11,14,21]
y=[1,7,12,15,22]
slope,intercept,r,p,str_err=stats.linregress(x,y)

def myfunc(x):
    return slope*x+intercept

mymodel=list(map(myfunc,x))
plt.scatter(x,y)
plt.plot(x,mymodel)
plt.show()
print("Slope =",slope)
print("Intercept =",intercept)
print("r =",r)
print("p =",p)
print("str_err =",str_err)

Slope = 1.0
Intercept = 1.0
r = 1.0
p = 1.2004217548761408e-30
str_err = 0.0
